link_to_help()

By default, uses the current controller and action to render the url to the corresponding RubyCampus wiki page.

    # File app/helpers/application_helper.rb, line 57
57:   def link_to_help
58:     wiki_page = "#{RUBYCAMPUS_ORG_BASE_URL}wiki/#{RUBYCAMPUS}/"
59:     case
60:       when controller.action_name == "index"   
61:         wiki_page << "Managing_#{controller.controller_name.titleize}"
62:       when controller.action_name == "show"   
63:         wiki_page << "Viewing_An_Existing_#{controller.controller_name.singularize.titleize}"
64:       when controller.action_name == "new"   
65:         wiki_page << "Creating_A_New_#{controller.controller_name.singularize.titleize}"
66:       when controller.action_name == "edit"   
67:         wiki_page << "Editing_An_Existing_#{controller.controller_name.singularize.titleize}"
68:       else
69:         wiki_page << "#{controller.controller_name.titleize}"
70:       end
71:     link_to _("Help"), "#{wiki_page}", :popup => true
72:   end
